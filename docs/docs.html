Core workflow

Python
import numpy as np
from latentmesh import LatentLLM, LatentGraph

llm = LatentLLM("deepseek-ai/DeepSeek-R1-Distill-Llama-8B")

flow = (
    LatentGraph(llm=llm, name="prod-flow", retrieval_k=3)
    .add_stage("plan", transform=lambda x: np.tanh(x))
    .add_stage("crit", transform=lambda x: np.maximum(0.0, x))
    .add_stage("write")
    .connect("plan", "crit")
    .connect("crit", "write")
    .compile(entry_stage="plan", exit_stage="write")
)

flow.add_example("How to rollback a failed deploy?", "Rollback to last healthy release and run smoke checks.")
print(flow.invoke("Migration failed in production. What is the first action?"))

Existing LangGraph RAG

Python
import numpy as np
from latentmesh import LatentLLM, LatentGraph

retriever = rag_vectorstore.as_retriever(search_kwargs={"k": 4})
llm = LatentLLM("meta-llama/Llama-3.1-8B-Instruct")

def rag_fuse(lat):
    docs = retriever.invoke("deploy rollback policy")
    ctx = "\n".join(d.page_content for d in docs)
    return (0.70 * lat) + (0.30 * llm.embed(ctx))

flow = (
    LatentGraph(llm=llm, name="rag-flow")
    .add_stage("retrieve", transform=rag_fuse)
    .add_stage("reason", transform=lambda x: np.tanh(x))
    .add_stage("write")
    .connect("retrieve", "reason")
    .connect("reason", "write")
    .compile(entry_stage="retrieve", exit_stage="write")
)
